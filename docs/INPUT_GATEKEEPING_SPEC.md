## 4. LAYER 2: VALIDATION

### 4.1 Input Quality Assessment

```typescript
/**
 * InputQualityAssessment
 *
 * Classifies input quality for downstream processing decisions
 */
interface InputQualityAssessment {
  /** Overall quality grade */
  grade: 'excellent' | 'good' | 'basic' | 'insufficient' | 'invalid';

  /** Numeric score 0-100 */
  score: number;

  /** Specific issues found */
  issues: InputIssue[];

  /** Can proceed with processing? */
  canProceed: boolean;

  /** Recommended action */
  action: 'proceed' | 'warn_and_proceed' | 'request_improvement' | 'block';

  /** Derived tone confidence (0-1) */
  toneConfidence: number;
}

interface InputIssue {
  field: string;
  type: 'too_short' | 'gibberish' | 'repetitive' | 'wrong_language' | 'spam';
  severity: 'warning' | 'error';
  message: string;
  suggestion?: string;
}

/**
 * assessInputQuality
 *
 * Main validation function
 */
function assessInputQuality(
  brandContext: BrandContext,
  culturalContext: CulturalContextInput
): InputQualityAssessment {
  const issues: InputIssue[] = [];
  let score = 100;

  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  // CHECK 1: Minimum Length
  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  const minLengths = getMinLengths(culturalContext.region);

  if (brandContext.targetAudience.length < minLengths.targetAudience) {
    score -= 20;
    issues.push({
      field: 'targetAudience',
      type: 'too_short',
      severity: 'warning',
      message: getLocalizedMessage('too_short', culturalContext.language),
      suggestion: getLocalizedSuggestion('targetAudience', culturalContext),
    });
  }

  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  // CHECK 2: Gibberish Detection
  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  if (isGibberish(brandContext.targetAudience)) {
    score -= 50;
    issues.push({
      field: 'targetAudience',
      type: 'gibberish',
      severity: 'error',
      message: getLocalizedMessage('gibberish', culturalContext.language),
    });
  }

  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  // CHECK 3: Language Consistency
  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€

  if (detectedLanguage !== culturalContext.language) {
    score -= 15;
    issues.push({
      field: 'targetAudience',
      type: 'wrong_language',
      severity: 'warning',
      message: getLocalizedMessage('wrong_language', culturalContext.language),
    });
  }

  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  // CHECK 4: Repetition/Spam
  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  if (isRepetitive(brandContext.targetAudience)) {
    score -= 30;
    issues.push({
      field: 'targetAudience',
      type: 'repetitive',
      severity: 'warning',
      message: getLocalizedMessage('repetitive', culturalContext.language),
    });
  }

  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  // COMPUTE FINAL ASSESSMENT
  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  const grade = scoreToGrade(score);
  const canProceed = score >= 30;
  const action = determineAction(grade);
  const toneConfidence = score / 100;

  return {
    grade,
    score,
    issues,
    canProceed,
    action,
    toneConfidence,
  };
}
```

```typescript
  // HEURISTIC 2: Keyboard Mash Detection
  // Adjacent keys pressed = "asdf", "qwer", "zxcv"
  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  const keyboardPatterns = [
    /[qwerty]{4,}/i,
    /[asdfgh]{4,}/i,
    /[zxcvbn]{4,}/i,
    /[12345]{4,}/,
    /(.)\1{3,}/,  // Repeated character: "aaaa"
  ];

  for (const pattern of keyboardPatterns) {
    if (pattern.test(input)) {
      return true;
    }
  }

  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  // HEURISTIC 3: No Real Words (for alphabetic languages)
  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  const isAlphabetic = /^[a-zA-Z\s]+$/.test(input);
  if (isAlphabetic) {
    const words = input.toLowerCase().split(/\s+/);
    const realWordCount = words.filter(w => COMMON_WORDS.has(w)).length;
    if (realWordCount === 0 && words.length > 2) {
      return true;  // No recognizable words
    }
  }

  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  // HEURISTIC 4: Chinese Character Validity
  // Random Unicode != valid Chinese
  // â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€â”€
  const hasChinese = /[\u4e00-\u9fff]/.test(input);
  if (hasChinese) {
    // Check if characters form valid sequences
    // (Advanced: use tokenizer to check if parseable)
    const validChineseRatio = countValidChineseSequences(input) / input.length;
    if (validChineseRatio < 0.5) {
      return true;
    }
  }

  return false;
}

/**
 * calculateEntropy
 *
 * Shannon entropy - high value = random/gibberish
 */
function calculateEntropy(str: string): number {
  const freq: Record<string, number> = {};
  for (const char of str) {
    freq[char] = (freq[char] || 0) + 1;
  }

  let entropy = 0;
  const len = str.length;
  for (const char in freq) {
    const p = freq[char] / len;
    entropy -= p * Math.log2(p);
  }

  return entropy;
}
```

## 7. COMPONENT SPECIFICATION

### 7.1 CulturalBrandContextForm

```typescript
/**
 * CulturalBrandContextForm
 *
 * Enhanced form with:
 * - Cultural placeholders and examples
 * - Real-time quality indicator
 * - Guided input chips
 * - Input validation
 * - Graceful degradation
 */
interface CulturalBrandContextFormProps {
  /** Cultural context for localization */
  culturalContext: CulturalContextInput;

  /** Callback when form values change */
  onChange: (context: BrandContext, quality: InputQualityAssessment) => void;

  /** Callback when form is submitted */
  onSubmit: (context: BrandContext, quality: InputQualityAssessment) => void;

  /** Initial values */
  initialValues?: Partial<BrandContext>;

  /** Show expanded fields by default */
  expandedByDefault?: boolean;

  /** Mode: minimal (2 fields) or full (6 fields) */
  mode?: 'minimal' | 'full';
}

/**
 * Component State
 */
interface FormState {
  values: BrandContext;
  quality: InputQualityAssessment;
  expanded: boolean;
  activeField: string | null;
}
```

### 7.2 InputQualityIndicator

```typescript
/**
 * InputQualityIndicator
 *
 * Real-time quality visualization
 */
interface InputQualityIndicatorProps {
  /** Current quality assessment */
  quality: InputQualityAssessment;

  /** Cultural context for localized messages */
  culturalContext: CulturalContextInput;

  /** Compact mode (just bar) or full (bar + suggestions) */
  mode?: 'compact' | 'full';
}
```

### 7.3 GuidedInputChips

```typescript
/**
 * GuidedInputChips
 *
 * Clickable chips that populate input field

interface GuidedInputChipsProps {
  /** Field this is attached to */
  field: 'targetAudience' | 'painPoints' | 'scenarios' | 'ctaOffer';

  /** Cultural context for localized chips */
  culturalContext: CulturalContextInput;

  /** Callback when chip is clicked */
  onChipClick: (value: string) => void;

  /** Currently selected chips */
  selectedChips?: string[];

  /** Max chips to show */
  maxVisible?: number;
}
```
